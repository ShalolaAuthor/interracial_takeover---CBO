bno_cure_disease_interaction = {
    icon = icon_bno_cure_disease
    category = interaction_category_friendly
    interface_priority = 999

    cooldown = { months = 1 }
    common_interaction = yes

    desc = {
        triggered_desc = {
            trigger = {
                scope:recipient = {
                    is_a_girl = yes
                }
            }
            desc = bno_cure_disease_interaction.desc.female
        }
        triggered_desc = {
            trigger = {
                scope:recipient = {
                    is_a_male = yes
                }
            }
            desc = bno_cure_disease_interaction.desc.male
        }
    }
    use_diplomatic_range = yes

    is_shown = {
        scope:actor = {
            bno_has_bbc_priest_or_priestess = yes
        }
    }

    is_valid_showing_failures_only = {
        scope:actor = {
            is_at_war = no
        }
        scope:recipient = {
            is_at_war = no
            custom_tooltip = {
		    	text = bno_cure_disease_interaction.recipient.sickness
		    	bno_illness_list = yes
		    }
            trigger_if = {
                limit = {
                    is_a_male = yes
                }
                is_bbc = yes
                custom_tooltip = {
                    text = bno_cure_disease_interaction.no_priestess
                    scope:actor = {
                        any_court_position_holder = {
                            type = bno_bbc_priestess_court_position
                        }
                    }
                }
            }
            trigger_else = {
                custom_tooltip = {
                    text = bno_cure_disease_interaction.no_priest
                    scope:actor = {
                        any_court_position_holder = {
                            type = bno_bbc_priest_court_position
                        }
                    }
                }
            }
        }
    }

    cost = {
        gold = {
            value = {
                add = scope:actor.medium_gold_value
                multiply = scope:recipient.health
                multiply = 0.5
            }
        }
    }

    on_accept = {
        show_as_tooltip = {
            if = {
                limit = {
                    scope:recipient = { is_a_girl = yes }
                }
                custom_tooltip = bno_cure_disease_interaction.female_recipient.effect
            }
            else_if = {
                limit = {
                    scope:recipient = { is_a_male = yes }
                }
                custom_tooltip = bno_cure_disease_interaction.male_recipient.effect
            }
        }
        hidden_effect = {
            scope:actor = {
                trigger_event = {
                    id = bno_embrace_african_5.0010
                    delayed = yes
                    days = { 7 14 }
                }
            }
        }
    }

    ai_potential = {
        OR = {
            any_child = {
                bno_illness_list = yes
            }
            any_spouse = {
                bno_illness_list = yes
            }
            any_relation = {
                type = lover
                bno_illness_list = yes
            }
            any_relation = {
                type = friend
                bno_illness_list = yes
            }
        }
    }
    
    ai_frequency = 12
    ai_instant_response = yes

    ai_targets = {
        ai_recipients = spouses
        ai_recipients = family
        ai_recipients = children
        ai_recipients = dynasty
        ai_recipients = scripted_relations
        max = 1
    }

    ai_will_do = {
        base = 100
    }
}
