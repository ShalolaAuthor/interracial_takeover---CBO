		custom_tooltip = {
			text = bno_events00.too_much_bbc

				is_adult = yes
				is_bbc = yes

		}

		months = 12
	}

			id = bno_events.0001
			days = 7
		}
	}

	ai_potential = {

			is_bbc = yes

				}
			}
		}
	}
	
	cost = {
		gold = 300
	}
	cooldown = {
		months = 12
	}
	effect = {
		send_interface_toast = {
			type = event_toast_effect_good
			title = bno_build_african_camps_toast
			scope:barony ?= {
				title_province = {
					add_special_building_slot  = bbc_camp_01
					add_special_building = bbc_camp_01
				}
			}
		}
	}
	ai_check_interval = 12
	ai_potential = {
		is_ruler = yes
		any_held_county = {
			any_in_de_jure_hierarchy = {
                bno_is_barony_valid_for_bbc_camps = {
					HOLDER = root
				}
			}
		}
	}
	ai_will_do = {
        base = 100
        modifier = {
			add = -50
			any_held_county = {
				any_in_de_jure_hierarchy = {
					title_province = {
						has_building_or_higher = bbc_camp_01

				}

#Convert to african decision
bno_convert_to_african = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_destiny_goal.dds"
	}
	sort_order = 10
	desc = bno_convert_to_african_desc
	selection_tooltip = bno_convert_to_african_tooltip
	decision_group_type = bno_decision_group
	is_shown = {
		is_adult = yes
		OR = {
			has_trait = bno_bbc
			has_trait = bno_whiteboi
			has_trait = bno_whitegirl
		}
		NOT = { is_culture_bnoc = yes }
	}
	is_valid = {
		is_valid_for_converting_to_african = yes
	}

	effect = {
		if = {
			limit = {
				any_spouse ?= {
					is_culture_bnoc = yes
				}
			}
			every_spouse = {
				limit = {
					is_culture_bnoc = yes
				}
				culture = { save_scope_as = spouse_culture }
			}
			set_culture = scope:spouse_culture
		}
		else_if = {
			limit = {
				any_courtier_or_guest = {
					is_culture_bnoc = yes
				}
			}
			random_courtier_or_guest = {
				limit = {
					is_culture_bnoc = yes
				}
				culture = {
					save_scope_as = new_culture
				}
			}
			set_culture = scope:new_culture
		}
	}
	ai_check_interval = 1
	ai_potential = {
		is_valid_for_converting_to_african = yes
	}
	ai_will_do = {
        base = 0
        bno_decision_modifier = yes
	}
}
