bno_build_bbc_church = {
    #Place Holder
    picture = { 
        reference = "gfx/interface/illustrations/decisions/decision_major_religion.dds"
    }
    desc = bno_build_bbc_church.desc
    selection_tooltip = bno_build_bbc_church.tooltip
    decision_group_type = bno_decision_group

    ai_check_interval = 1

    is_shown = {
        bno_has_embrace_african_5 = yes
    }

    is_valid = {
        is_independent_ruler = yes
        any_held_county = {
			any_in_de_jure_hierarchy = {
                bno_is_barony_valid_for_church_foundation = {
                    HOLDER = root
                }
			}
		}
        custom_description = {
            text = is_king_or_emperor
            highest_held_title_tier >= tier_kingdom
        }
        piety_level >= 3
        is_faith_bbcs = yes
    }
    is_valid_showing_failures_only = {
        is_available_adult = yes
        is_at_war = no
    }

    cooldown = {
		months = 12
	}
    widget = {
        controller = create_holy_order
        barony_valid = {
            trigger_if = {
                limit = { exists = this }
                bno_is_barony_valid_for_church_foundation = {
                    HOLDER = scope:ruler
                }
            }
            trigger_else = {
                custom_description = { 
                    text = has_holy_order_barony_selected 
                    always = no 
                }
            }
        }
    }

    cost = {
        #can change later if too cheap
        gold = { value = holy_order_gold_cost  }
        piety = { value = holy_order_piety_cost }
    }
    effect = {
        send_interface_toast = {
            type = event_toast_effect_good
            title = bno_build_bbc_church.toast
            scope:barony ?= {
                title_province = {
                    add_special_building_slot = bbc_church_01
                    add_special_building = bbc_church_01
                }
            }
        }
    }

    ai_potential = {
        highest_held_title_tier >= tier_kingdom
        bno_has_embrace_african_5 = yes
        is_playable_character = yes
        any_held_county = {
			any_in_de_jure_hierarchy = {
                bno_is_barony_valid_for_church_foundation = {
                    HOLDER = root
                }
			}
		}
    }

    ai_will_do = {
        base = 1000
        modifier = {
            add = -950
            any_held_county = {
				any_in_de_jure_hierarchy = {
					title_province = {
						has_building_or_higher = bbc_church_01
					}
				}
			}
        }
    }
}
