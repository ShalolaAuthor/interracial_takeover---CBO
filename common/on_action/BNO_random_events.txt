bno_religion_maintenance = {
	trigger = {
		has_game_rule = bno_mod_status_on
		has_game_rule = bno_religion_yes
	}
	effect = {
		if = {
			limit = {
				is_ai = yes
				is_bbc = yes
				any_religion_global = {
					any_faith = {
						NOT = { religious_head = root }
					}
				}
				NOT = {
					faith = faith:bbcs
				}
			}
			set_character_faith = faith:bbcs
		}
	}
}
