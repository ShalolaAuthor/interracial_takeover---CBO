bno_non_playable_pulse = {
	on_actions = {
		bno_non_playable_yearly_initial_pulse
		bno_non_playable_yearly_global_random_pulse
	}
}

bno_non_playable_yearly_initial_pulse = {
	trigger = {
		is_ai = no
		is_elligible_partner = yes
	}
	events = {
		bno_non_playable_events.0001
	}
}

bno_yearly_non_playable_pulse = {
	random_events = {
		chance_to_happen = 100
		chance_of_no_event = {
			value = 0
			if = {
				limit = { is_ai = yes }
				add = bno_ai_events_chance
			}
		}
		1 = bno_non_playable.0001 # landless bbc will steal playable gold
	}
}

bno_non_playable_yearly_global_random_pulse = {
	trigger = {
		has_game_rule = bno_global_bbc_activities_yes
	}
	on_actions = {
		bno_yearly_non_playable_pulse
	}
}
