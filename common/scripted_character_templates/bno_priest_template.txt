bno_male_priest_character_template = {
	age = bno_younger_adult
	random_traits_list = {
		count = 1
		education_learning_2 = {}
		education_learning_3 = {}
		education_learning_4 = {}
	}

	trait = bno_full_black
	trait = bno_lifestyle_bbc_priest
	random_traits = yes
	
	faith = faith:bbcs
	culture = culture:bnoc
	gender = male

	learning = {
		min_template_high_skill
		max_template_high_skill
	}

	after_creation = {
		add_character_flag = need_priest_outfit
		bno_get_between_common_uncommon_and_rare_trait = {
			TRAIT_COMMON = bno_snowbunny
			TRAIT_UNCOMMON = bno_qos
			TRAIT_RARE = bno_mindless_bbc_slut
		}
		bno_maybe_get_a_rare_trait = {
			TRAIT = bno_pure_white
		}
        bno_get_a_leveled_trait = {
			TRAIT = beauty_good
		}
		bno_add_bbc_trait_effect = yes
		if = {
			limit = {
				root = {
					is_faith_bbcs = yes
				}
			}
			set_character_faith = root.faith
		}
		if = {
			limit = {
				root = {
					is_culture_bnoc = yes
				}
			}
			set_culture = root.culture
		}
        remove_bad_traits = yes
	}
}
