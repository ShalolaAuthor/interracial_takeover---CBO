bno_is_county_not_fully_controlled = {
    any_held_title = {
        tier = tier_county
        is_landless_type_title = no
        county_control < 100
    }
}

bno_does_local_bbc_exist = {
    any_realm_province = {
        any_pool_character = {
            province = prev
            is_blackbull = yes
        }
    }
}

bno_illness_list = {
	OR = {
		has_trait = lovers_pox
		has_trait = ill
		has_trait = pneumonic
		has_trait = great_pox
		has_trait = early_great_pox
		has_trait = leper
		has_trait = gout_ridden
		has_trait = consumption
		has_trait = cancer
		has_trait = typhus
		has_trait = bubonic_plague
		has_trait = smallpox
		has_trait = measles
		has_trait = dysentery
		has_trait = ergotism
		has_trait = sickly
	}
}

bno_has_embrace_african_5 = {
	dynasty ?= {
		has_dynasty_perk = bno_embrace_african_5
	}
}

bno_has_embrace_african_3 = {
	dynasty ?= {
		has_dynasty_perk = bno_embrace_african_3
	}
}

bno_has_bbc_priest_or_priestess = {
	OR = {
		any_court_position_holder = {
			type = bno_bbc_priest_court_position
		}
		any_court_position_holder = {
			type = bno_bbc_priestess_court_position
		}
	}
}

bno_is_barony_valid_for_church_foundation = {
	is_capital_barony = no
	title_province = {
        has_special_building_slot = no
    }
	county = {
		trigger_if = {
			limit = { exists = scope:ruler }
			holder = scope:ruler
			this = scope:ruler.capital_county
		}
		trigger_else = {
			holder = $HOLDER$
			this = $HOLDER$.capital_county
		}
	}
    is_leased_out = no
    NOT = { title_province = { has_building_or_higher = bbc_church_01 } }
}

bno_is_barony_valid_for_culture_change = {
	trigger_if = {
		limit = {
			exists = scope:ruler
		}
		holder = scope:ruler
	}
    trigger_else = {
		holder = root
	}
	county = {
		county_control >= 100
	}
	title_province = {
		is_culture_bnoc = no
		is_county_capital = yes
	}
}
