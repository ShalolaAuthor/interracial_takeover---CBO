#pop when child is 15
bno_ritual_of_purification.1111 = {
    hidden = yes
    trigger = {
        age >= 15
        is_a_girl = yes
        is_ai = yes
        has_trait = bno_blood
        NOT = { has_character_flag = bno_ritual_of_purification.1111 }
        OR = {
            mother ?= {
                is_playable_character = yes
                is_faith_bbcs = yes
                is_bno_blacked_and_up = yes
                any_councillor = {
                    has_council_position = councillor_court_chaplain
                }
            }
            real_father ?= {
                is_playable_character = yes
                is_faith_bbcs = yes
                is_bbc = yes
                any_councillor = {
                    has_council_position = councillor_court_chaplain
                }
            }
        }
    }
    immediate = {
        save_scope_as = child
        add_character_flag = bno_ritual_of_purification.1111
        if = {
            limit = {
                mother ?= {
                    is_playable_character = yes
                    is_faith_bbcs = yes
                    is_bno_blacked_and_up = yes
                    any_councillor = {
                        has_council_position = councillor_court_chaplain
                    }
                }
            }
            mother = {
                trigger_event = bno_ritual_of_purification.0001
            }
        }
        else_if = {
            limit = {
                real_father ?= {
                    is_playable_character = yes
                    is_faith_bbcs = yes
                    is_bbc = yes
                    any_councillor = {
                        has_council_position = councillor_court_chaplain
                    }
                }
            }
            real_father = {
                trigger_event = bno_ritual_of_purification.0001
            }
        }
    }
}
